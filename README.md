# installation

    cd ~
    git clone git@github.com:petermallett/conf.git
    ./conf/bin/install    

# Vim

    run vim, then use the command:
    :PlugInstall

# Bash

    create ~/.bash_private
    copy existing settings, if needed

# Tmux

    git clone https://github.com/tmux-plugins/tpm ~/.tmux/plugins/tpm
    install plugins with <prefix> + I
